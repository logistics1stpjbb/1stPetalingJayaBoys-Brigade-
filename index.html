📊 Google Sheets Sync
When attendance is saved, it will POST to a Google Apps Script Web App that writes to your sheet. Paste the script URL below.
Found between /d/ and /edit in your Google Sheets URL
Apps Script Code — paste into script.google.com → Deploy as Web App (Anyone)
// ═══════════════════════════════════════════════════════════════════
// Boys Brigade Attendance Sync — Google Apps Script
// Deploy as Web App: Execute as Me, Who has access: Anyone
// ═══════════════════════════════════════════════════════════════════
//
// Band Attendance 2026:  Row2=dates("11/4"), Row3=sub-headers, ColA=Name, ColB=Section, ColC=Squad, data from ColF
// Squad sheets:          Row6=dates("11th April 2026"), Row8=headers, ColD=Name, data from ~ColN

function doPost(e) {
  try {
    var data          = JSON.parse(e.postData.contents);
    var ss            = SpreadsheetApp.openById(data.sheetId);
    var records       = data.records  || [];
    var weekLabel     = data.weekLabel     || '';
    var weekDateShort = data.weekDateShort || '';   // "11/4"   — band sheet
    var weekDateLong  = data.weekDateLong  || '';   // "11th April 2026" — squad sheets
    // Fallbacks if client is old
    if (!weekDateShort && data.weekDate) weekDateShort = data.weekDate;
    if (!weekDateLong  && data.weekDate) weekDateLong  = data.weekDate;

    var bandSheet  = ss.getSheetByName('Band Attendance 2026');
    var squadSheets = {
      'Alpha': ss.getSheetByName('Squad Alpha'),
      'Papa' : ss.getSheetByName('Squad Papa'),
      'Zulu' : ss.getSheetByName('Squad Zulu'),
      'Hotel': ss.getSheetByName('Squad Hotel')
    };

    // ── Cache: read full rows once per sheet ──────────────────────────────
    var cache = {};
    function getSheetCache(sh) {
      if (!sh) return null;
      var name = sh.getName();
      if (cache[name]) return cache[name];
      var lastR = sh.getLastRow(), lastC = sh.getLastColumn();
      var c = {
        lastR: lastR, lastC: lastC,
        row2:  lastC > 0 ? sh.getRange(2,  1, 1, lastC).getValues()[0] : [],
        row3:  lastC > 0 ? sh.getRange(3,  1, 1, lastC).getValues()[0] : [],
        row6:  lastC > 0 ? sh.getRange(6,  1, 1, lastC).getValues()[0] : [],
        row8:  lastC > 0 ? sh.getRange(8,  1, 1, lastC).getValues()[0] : [],
        nameA: lastR > 3  ? sh.getRange(4,  1, lastR-3, 1).getValues() : [],  // band col A
        nameD: lastR > 8  ? sh.getRange(9,  4, Math.max(1,lastR-8), 1).getValues() : [] // squad col D
      };
      cache[name] = c;
      return c;
    }

    // ── Name matching: website may have full name, sheet may have shorter name ──
    // "Joshua Ethan Yogendran" matches "Joshua Ethan"
    // "Jason Koh Chien Jed"   matches "Jason Koh"
    // Strategy: sheet name starts the website name, OR website name starts the sheet name.
    // We always prefer an exact match first, then fall back to prefix matching.
    function namesMatch(sheetName, websiteName) {
      var s = sheetName.trim().toLowerCase();
      var w = websiteName.trim().toLowerCase();
      if (s === w) return true;                          // exact
      if (w.indexOf(s) === 0 && w[s.length] === ' ') return true;  // sheet is prefix of web
      if (s.indexOf(w) === 0 && s[w.length] === ' ') return true;  // web is prefix of sheet
      return false;
    }

    // ── fix2: normalise a value — if it was never set, write 0; if set, keep it ──
    // The portal sends '' when untouched. We treat '' as 0 so cells are never blank.
    function v(val) {
      if (val === 1) return 1;
      if (val === 0) return 0;
      return 0;   // '' / undefined / null → 0
    }

    // ── Band sheet helpers ────────────────────────────────────────────────
    function findBandRow(c, name) {
      var exact = -1, prefix = -1;
      for (var i = 0; i < c.nameA.length; i++) {
        var cell = (c.nameA[i][0]||'').toString();
        if (namesMatch(cell, name)) {
          if (cell.trim().toLowerCase() === name.trim().toLowerCase()) return 4+i; // exact → done
          if (prefix === -1) prefix = 4+i;
        }
      }
      return prefix; // best prefix match, or -1
    }
    function findBandWeekCol(c, dateShort, label) {
      // Row2 has dates like "11/4", "07/03". Scan every column (merged cells mean the
      // value only sits in the first column of a merged group, so don't skip by 3).
      var needles = [dateShort, label].filter(Boolean).map(function(s){ return s.toLowerCase().trim(); });
      for (var i = 5; i < c.row2.length; i++) {
        var cell = (c.row2[i]||'').toString().toLowerCase().trim();
        if (!cell) continue;
        for (var n = 0; n < needles.length; n++) {
          if (cell === needles[n] || cell.indexOf(needles[n]) !== -1) return i+1; // 1-based
        }
      }
      return -1;
    }

    // ── Squad sheet helpers ───────────────────────────────────────────────
    function findSquadRow(c, name) {
      var prefix = -1;
      for (var i = 0; i < c.nameD.length; i++) {
        var cell = (c.nameD[i][0]||'').toString();
        if (namesMatch(cell, name)) {
          if (cell.trim().toLowerCase() === name.trim().toLowerCase()) return 9+i;
          if (prefix === -1) prefix = 9+i;
        }
      }
      return prefix;
    }
    function findSquadWeekCol(c, dateLong, label) {
      // Row6 has dates like "11th April 2026". Match dateLong first, then label.
      var needles = [dateLong, label].filter(Boolean).map(function(s){ return s.toLowerCase().trim(); });
      for (var i = 0; i < c.row6.length; i++) {
        var cell = (c.row6[i]||'').toString().toLowerCase().trim();
        if (!cell) continue;
        for (var n = 0; n < needles.length; n++) {
          if (cell.indexOf(needles[n]) !== -1) return i+1;
        }
      }
      return -1;
    }
    function mapSquadCols(c, startCol) {
      // scan row8 from startCol until next week block starts in row6
      var m = {};
      for (var i = startCol-1; i < c.lastC; i++) {
        if (i > startCol-1 && (c.row6[i]||'').toString().trim() !== '') break; // next week
        var h = (c.row8[i]||'').toString().toUpperCase().trim();
        if      (h.indexOf('BAND')   !== -1 && h.indexOf('OPENING')   !== -1) m.bandOpen     = i+1;
        else if (h.indexOf('BAND')   !== -1 && h.indexOf('DISMISSAL') !== -1) m.bandDismiss  = i+1;
        else if (h.indexOf('BAND')   !== -1 && h.indexOf('LEAVE')     !== -1) m.bandLeave    = i+1;
        else if (h.indexOf('PARADE') !== -1 && h.indexOf('OPENING')   !== -1) m.paradeOpen   = i+1;
        else if (h.indexOf('PARADE') !== -1 && h.indexOf('DISMISSAL') !== -1) m.paradeDismiss= i+1;
        else if (h.indexOf('UNIFORM')!== -1)                                   m.uniform      = i+1;
        else if (h.indexOf('LEAVE')  !== -1 && h.indexOf('PARADE')    !== -1)  m.paradeLeave  = i+1;
        else if (h.indexOf('REMARK') !== -1)                                    m.remarks      = i+1;
      }
      return m;
    }

    // ── Batch write accumulator ───────────────────────────────────────────
    var writes = {};
    function queueWrite(sh, row, col, val) {
      // Allow 0 — only skip if column not found or row invalid
      if (!sh || !col || col < 1 || row < 1) return;
      var key = sh.getName() + ':' + row;
      if (!writes[key]) writes[key] = { sh: sh, row: row, cols: {} };
      writes[key].cols[col] = val;
    }
    function flushWrites() {
      Object.keys(writes).forEach(function(key) {
        var entry = writes[key];
        var cols = Object.keys(entry.cols).map(Number).sort(function(a,b){return a-b;});
        if (!cols.length) return;
        var minC = cols[0], maxC = cols[cols.length-1];
        // Read existing row range, overlay our values, then setValues in one call
        var range = entry.sh.getRange(entry.row, minC, 1, maxC-minC+1);
        var rowVals = range.getValues()[0];
        cols.forEach(function(c) { rowVals[c - minC] = entry.cols[c]; });
        range.setValues([rowVals]);
      });
    }

    // ── Pre-compute band and squad col mappings (one per sheet) ───────────
    var bandWeekCol = -1, bandC = null;
    if (bandSheet) {
      bandC = getSheetCache(bandSheet);
      bandWeekCol = findBandWeekCol(bandC, weekDateShort, weekLabel);
    }
    var squadColMaps = {};
    Object.keys(squadSheets).forEach(function(sq) {
      var sh = squadSheets[sq]; if (!sh) return;
      var c = getSheetCache(sh);
      var startCol = findSquadWeekCol(c, weekDateLong, weekLabel);
      squadColMaps[sq] = startCol > 0 ? { c: c, sh: sh, startCol: startCol, cols: mapSquadCols(c, startCol) } : null;
    });

    // ── Process each record ───────────────────────────────────────────────
    records.forEach(function(r) {
      // 1. Band sheet  (col layout: weekCol=Valid Leave, weekCol+1=Opening, weekCol+2=Closing)
      if (bandSheet && bandWeekCol > 0) {
        var bRow = findBandRow(bandC, r.name);
        if (bRow > 0) {
          queueWrite(bandSheet, bRow, bandWeekCol,     v(r.bandLeave));   // Valid Leave
          queueWrite(bandSheet, bRow, bandWeekCol + 1, v(r.bandOpen));    // Opening
          queueWrite(bandSheet, bRow, bandWeekCol + 2, v(r.bandDismiss)); // Closing
        }
      }
      // 2. Squad sheet
      var sq = r.squad || '';
      var sqMap = squadColMaps[sq];
      if (sqMap && sqMap.startCol > 0) {
        var sqRow = findSquadRow(sqMap.c, r.name);
        if (sqRow > 0) {
          var m = sqMap.cols;
          queueWrite(sqMap.sh, sqRow, m.bandOpen,      v(r.bandOpen));
          queueWrite(sqMap.sh, sqRow, m.bandDismiss,   v(r.bandDismiss));
          queueWrite(sqMap.sh, sqRow, m.bandLeave,     v(r.bandLeave));
          queueWrite(sqMap.sh, sqRow, m.paradeOpen,    v(r.paradeOpen));
          queueWrite(sqMap.sh, sqRow, m.paradeDismiss, v(r.paradeDismiss));
          queueWrite(sqMap.sh, sqRow, m.uniform,       v(r.uniform));
          queueWrite(sqMap.sh, sqRow, m.paradeLeave,   v(r.paradeLeave));
          if (m.remarks && r.remarks && r.remarks.toString().trim())
            queueWrite(sqMap.sh, sqRow, m.remarks, r.remarks);
        }
      }
    });

    // Write all collected values in bulk (much faster than cell-by-cell)
    flushWrites();

    return ContentService
      .createTextOutput(JSON.stringify({ok: true}))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok: false, error: err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
